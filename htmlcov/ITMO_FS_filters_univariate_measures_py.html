52 def count_hits(feature): 

53 splits = {cl: feature[y == cl] for cl in classes} 

54 means = {cl: np.mean(splits[cl]) for cl in classes} 

55 devs = {cl: np.var(splits[cl]) for cl in classes} 

56 distances = np.vectorize( 

57 lambda x_val: {cl: (

97 mean_feature = np.mean(feature)

168 features due to being based on information theory. 

169

171 ---------- 

172 x : array-like, shape (n_samples, n_features)

198 def __SU(feature): 

199 entropy_x = entropy(feature) 

200 return (2 * (entropy_x - conditional_entropy(y, feature))

208def kendall_corr(x, y): 

209 """Calculate Sample sign correlation (Kendall correlation) for each 

210 feature. Bigger absolute values mean more important features. 

211 

212 Parameters 

213 ---------- 

214 x : array-like, shape (n_samples, n_features) 

215 The training input samples. 

216 y : array-like, shape (n_samples,) 

217 The target values. 

218 

219 Returns 

220 ------- 

221 array-like, shape (n_features,) : feature scores 

222 

223 See Also 

224 -------- 

225 https://en.wikipedia.org/wiki/Kendall_rank_correlation_coefficient 

226 

227 Examples 

228 -------- 

229 >>> from ITMO_FS.filters.univariate import kendall_corr 

230 >>> import numpy as np 

231 >>> x = np.array([[3, 3, 3, 2, 2], [3, 3, 1, 2, 3], [1, 3, 5, 1, 1], 

232 ... [3, 1, 4, 3, 1], [3, 1, 2, 3, 1]]) 

233 >>> y = np.array([1, 3, 2, 1, 2]) 

234 >>> kendall_corr(x, y) 

235 array([-0.1, 0.2, -0.4, -0.2, 0.2]) 

236 """ 

237 def __kendall_corr(feature): 

238 k_corr = 0.0 

239 for i in range(len(feature)): 

240 k_corr += np.sum(np.sign(feature[i] - feature[i + 1:]) 

241 * np.sign(y[i] - y[i + 1:])) 

242 return 2 * k_corr / (feature.shape[0] * (feature.shape[0] - 1)) 

243 

244 return np.apply_along_axis(__kendall_corr, 0, x)

279def reliefF_measure(x, y, k_neighbors=1): 

280 """Calculate ReliefF measure for each feature. Bigger values mean more 

281 important features. 

282 

283 Note: 

284 Only for complete x 

285 Rather than repeating the algorithm m(TODO Ask Nikita about user defined) 

286 times, implement it exhaustively (i.e. n times, once for each instance) 

287 for relatively small n (up to one thousand). 

288 

289 Calculates spearman correlation for each feature. 

290 Spearman's correlation assesses monotonic relationships (whether linear or 

291 not). If there are no repeated data values, a perfect Spearman correlation 

292 of +1 or −1 occurs when each of the variables is a perfect monotone 

293 function of the other. 

294 

295 Parameters 

296 ---------- 

297 x : array-like, shape (n_samples, n_features) 

298 The input samples. 

299 y : array-like, shape (n_samples,) 

300 The classes for the samples. 

301 k_neighbors : int, optional 

302 The number of neighbors to consider when assigning feature importance 

303 scores. More neighbors results in more accurate scores but takes 

304 longer. Selection of k hits and misses is the basic difference to 

305 Relief and ensures greater robustness of the algorithm concerning noise. 

306 

307 Returns 

308 ------- 

309 array-like, shape (n_features,) : feature scores 

310 

311 See Also 

312 -------- 

313 R.J. Urbanowicz et al. Relief-based feature selection: Introduction and 

314 review. Journal of Biomedical Informatics 85 (2018) 189–203 

315 

316 Examples 

317 -------- 

318 >>> from ITMO_FS.filters.univariate import reliefF_measure 

319 >>> import numpy as np 

320 >>> x = np.array([[3, 3, 3, 2, 2], [3, 3, 1, 2, 3], [1, 3, 5, 1, 1], 

321 ... [3, 1, 4, 3, 1], [3, 1, 2, 3, 1], [1, 2, 1, 4, 2], [4, 3, 2, 3, 1]]) 

322 >>> y = np.array([1, 2, 2, 1, 2, 1, 2]) 

323 >>> reliefF_measure(x, y) 

324 array([-0.14285714, -0.57142857, 0.10714286, -0.14285714, 0.07142857]) 

325 >>> reliefF_measure(x, y, k_neighbors=2) 

326 array([-0.07142857, -0.17857143, -0.07142857, -0.0952381 , -0.17857143]) 

327 """ 

328 def __calc_misses(index): 

329 misses_diffs_classes = np.abs( 

330 np.vectorize( 

331 lambda cl: ( 

332 x[index] 

333 - x[knn_from_class(dm, y, index, k_neighbors, cl)]) 

334 * prior_prob[cl], 

335 signature='()->(n,m)')(classes[classes != y[index]])) 

336 return (np.sum(np.sum(misses_diffs_classes, axis=1), axis=0) 

337 / (1 - prior_prob[y[index]])) 

338 

339 classes, counts = np.unique(y, return_counts=True) 

340 if np.any(counts <= k_neighbors): 

341 raise ValueError( 

342 "Cannot calculate relieff measure because one of theclasses has " 

343 "less than %d samples" % (k_neighbors + 1)) 

344 prior_prob = dict(zip(classes, np.array(counts) / len(y))) 

345 n_samples = x.shape[0] 

346 n_features = x.shape[1] 

347 # use manhattan distance instead of euclidean 

348 dm = pairwise_distances(x, x, 'manhattan') 

349 

350 indices = np.arange(n_samples) 

351 # use abs instead of square because of manhattan distance 

352 hits_diffs = np.abs( 

353 np.vectorize( 

354 lambda index: ( 

355 x[index] 

356 - x[knn_from_class(dm, y, index, k_neighbors, y[index])]), 

357 signature='()->(n,m)')(indices)) 

358 H = np.sum(hits_diffs, axis=(0,1)) 

359 

360 misses_sum_diffs = np.vectorize( 

361 lambda index: __calc_misses(index), 

362 signature='()->(n)')(indices) 

363 M = np.sum(misses_sum_diffs, axis=0) 

364 

365 weights = M - H 

366 # dividing by m * k guarantees that all final weights 

367 # will be normalized within the interval [ − 1, 1]. 

368 weights /= n_samples * k_neighbors 

369 # The maximum and minimum values of A are determined over the entire 

370 # set of instances. 

371 # This normalization ensures that weight updates fall 

372 # between 0 and 1 for both discrete and continuous features. 

373 with np.errstate(divide='ignore', invalid="ignore"): # todo 

374 return weights / (np.amax(x, axis=0) - np.amin(x, axis=0))

377def relief_measure(x, y, m=None, random_state=42): 

378 """Calculate Relief measure for each feature. This measure is supposed to 

379 work only with binary classification datasets; for multi-class problems use 

380 the ReliefF measure. Bigger values mean more important features. 

381 

382 Parameters 

383 ---------- 

384 x : array-like, shape (n_samples, n_features) 

385 The input samples. 

386 y : array-like, shape (n_samples,) 

387 The classes for the samples. 

388 m : int, optional 

389 Amount of iterations to do. If not specified, n_samples iterations 

390 would be performed. 

391 random_state : int, optional 

392 Random state for numpy random. 

393 

394 Returns 

395 ------- 

396 array-like, shape (n_features,) : feature scores 

397 

398 See Also 

399 -------- 

400 R.J. Urbanowicz et al. Relief-based feature selection: Introduction and 

401 review. Journal of Biomedical Informatics 85 (2018) 189–203 

402 

403 Examples 

404 -------- 

405 >>> from ITMO_FS.filters.univariate import relief_measure 

406 >>> import numpy as np 

407 >>> x = np.array([[3, 3, 3, 2, 2], [3, 3, 1, 2, 3], [1, 3, 5, 1, 1], 

408 ... [3, 1, 4, 3, 1], [3, 1, 2, 3, 1]]) 

409 >>> y = np.array([1, 2, 2, 1, 2]) 

410 >>> relief_measure(x, y) 

411 array([ 0. , -0.6 , -0.1875, -0.15 , -0.4 ]) 

412 """ 

413 weights = np.zeros(x.shape[1]) 

414 classes, counts = np.unique(y, return_counts=True) 

415 if len(classes) == 1: 

416 raise ValueError("Cannot calculate relief measure with 1 class") 

417 if 1 in counts: 

418 raise ValueError( 

419 "Cannot calculate relief measure because one of the classes has " 

420 "only 1 sample") 

421 

422 n_samples = x.shape[0] 

423 n_features = x.shape[1] 

424 if m is None: 

425 m = n_samples 

426 

427 x_normalized = MinMaxScaler().fit_transform(x) 

428 dm = euclidean_distances(x_normalized, x_normalized) 

429 indices = np.random.default_rng(random_state).integers( 

430 low=0, high=n_samples, size=m) 

431 objects = x_normalized[indices] 

432 hits_diffs = np.square( 

433 np.vectorize( 

434 lambda index: ( 

435 x_normalized[index] 

436 - x_normalized[knn_from_class(dm, y, index, 1, y[index])]), 

437 signature='()->(n,m)')(indices)) 

438 misses_diffs = np.square( 

439 np.vectorize( 

440 lambda index: ( 

441 x_normalized[index] 

442 - x_normalized[knn_from_class( 

443 dm, y, index, 1, y[index], anyOtherClass=True)]), 

444 signature='()->(n,m)')(indices)) 

445 

446 H = np.sum(hits_diffs, axis=(0,1)) 

447 M = np.sum(misses_diffs, axis=(0,1)) 

448 

449 weights = M - H 

450 

451 return weights / m

457 to being based on statistics. 

458 

459 Parameters 

460 ---------- 

461 x : array-like, shape (n_samples, n_features) 

462 The training input samples. 

463 y : array-like, shape (n_samples,) 

464 The target values. 

465 

466 Returns

468 array-like, shape (n_features,) : feature scores

486 """ 

487 def __chi2(feature): 

488 values, counts = np.unique(feature, return_counts=True) 

489 values_map = {val: idx for idx, val in enumerate(values)} 

490 splits = {cl: np.array([values_map[val] for val in feature[y == cl]]) 

491 for cl in classes}

493 lambda cl: prior_probs[cl] * counts, 

494 signature='()->(1)')(classes) 

495 n = np.vectorize( 

496 lambda cl: np.bincount(splits[cl], minlength=values.shape[0]), 

497 signature='()->(1)')(classes) 

498 return np.sum(np.square(e - n) / e) 

499 

500 classes, counts = np.unique(y, return_counts=True) 

501 prior_probs = {cl: counts[idx] / x.shape[0] for idx, cl 

502 in enumerate(classes)} 

503 

504 return np.apply_along_axis(__chi2, 0, x)

588 >>> spearman_corr(x, y) 

589 array([-0.186339 , 0.30429031, -0.52704628, -0.30555556, 0.35355339]) 

590 """ 

591 n = x.shape[0] 

592 if n < 2: 

593 raise ValueError("The input should contain more than 1 sample") 

594 

595 x_ranks = np.apply_along_axis(rankdata, 0, x) 

596 y_ranks = rankdata(y) 

597

601def pearson_corr(x, y): 

602 """Calculate Pearson's correlation for each feature. Bigger absolute 

603 values mean more important features. This measure works best with discrete 

604 features due to being based on statistics. 

605 

606 Parameters 

607 ---------- 

608 x : array-like, shape (n_samples, n_features) 

609 The training input samples. 

610 y : array-like, shape (n_samples,) 

611 The target values. 

612 

613 Returns 

614 ------- 

615 array-like, shape (n_features,) : feature scores 

616 

617 See Also 

618 -------- 

619 https://en.wikipedia.org/wiki/Pearson_correlation_coefficient 

620 

621 Examples 

622 -------- 

623 >>> from ITMO_FS.filters.univariate import pearson_corr 

624 >>> from sklearn.preprocessing import KBinsDiscretizer 

625 >>> import numpy as np 

626 >>> x = np.array([[3, 3, 3, 2, 2], [3, 3, 1, 2, 3], [1, 3, 5, 1, 1], 

627 ... [3, 1, 4, 3, 1], [3, 1, 2, 3, 1]]) 

628 >>> y = np.array([1, 3, 2, 1, 2]) 

629 >>> est = KBinsDiscretizer(n_bins=10, encode='ordinal') 

630 >>> x = est.fit_transform(x) 

631 >>> pearson_corr(x, y) 

632 array([-0.13363062, 0.32732684, -0.60631301, -0.26244533, 0.53452248]) 

633 """ 

634 x_dev = x - np.mean(x, axis=0) 

635 y_dev = y - np.mean(y) 

636 sq_dev_x = x_dev * x_dev 

637 sq_dev_y = y_dev * y_dev 

638 sum_dev = y_dev.T.dot(x_dev).reshape((x.shape[1],)) 

639 denominators = np.sqrt(np.sum(sq_dev_y) * np.sum(sq_dev_x, axis=0)) 

640 

641 results = np.array( 

642 [(sum_dev[i] / denominators[i]) if denominators[i] > 0.0 else 0 for i 

643 in range(len(denominators))]) 

644 return results

648def laplacian_score(x, y, k_neighbors=5, t=1, metric='euclidean', **kwargs): 

649 """Calculate Laplacian Score for each feature. Smaller values mean more 

650 important features. 

651 

652 Parameters 

653 ---------- 

654 x : array-like, shape (n_samples, n_features) 

655 The input samples. 

656 y : array-like, shape (n_samples,) 

657 The classes for the samples. 

658 k_neighbors : int, optional 

659 The number of neighbors to construct a nearest neighbor graph. 

660 t : float, optional 

661 Suitable constant for weight matrix S 

662 where Sij = exp(-(|xi - xj| ^ 2) / t). 

663 metric : str or callable, optional 

664 Norm function to compute distance between two points or one of the 

665 commonly used strings ('euclidean', 'manhattan' etc.) The default 

666 metric is euclidean. 

667 weights : array-like, shape (n_samples, n_samples) 

668 The weight matrix of the graph that models the local structure of 

669 the data space. By default it is constructed using KNN algorithm. 

670 

671 Returns 

672 ------- 

673 array-like, shape (n_features,) : feature scores 

674 

675 See Also 

676 -------- 

677 https://papers.nips.cc/paper/2909-laplacian-score-for-feature-selection.pdf 

678 

679 Examples 

680 -------- 

681 >>> from ITMO_FS.filters.univariate import laplacian_score 

682 >>> import numpy as np 

683 >>> x = np.array([[1, 2, 3, 3, 1], [2, 2, 3, 3, 2], [1, 3, 3, 1, 3], 

684 ... [3, 1, 3, 1, 4], [4, 4, 3, 1, 5]]) 

685 >>> y = np.array([1, 2, 3, 4, 5]) 

686 >>> laplacian_score(x, y) 

687 array([1.98983619, 1.22248371, nan, 0.79710221, 1.90648048]) 

688 """ 

689 n, m = x.shape 

690 k_neighbors = min(k_neighbors, n - 1) 

691 if 'weights' in kwargs.keys(): 

692 S = kwargs['weights'] 

693 else: 

694 if n > 100000: 

695 S = lil_matrix((n, n)) 

696 else: 

697 S = np.zeros((n, n)) 

698 graph = NearestNeighbors(n_neighbors=k_neighbors, metric=metric) 

699 graph.fit(x) 

700 distances, neighbors = graph.kneighbors() 

701 for i in range(n): 

702 for j in range(k_neighbors): 

703 S[i, neighbors[i][j]] = S[neighbors[i][j], i] = exp( 

704 -distances[i][j] * distances[i][j] / t) 

705 ONE = np.ones((n,)) 

706 D = np.diag(S.dot(ONE)) 

707 L = D - S 

708 t = D.dot(ONE) 

709 F = x - x.T.dot(t) / ONE.dot(t) 

710 F = F.T.dot(L.dot(F)) / F.T.dot(D.dot(F)) 

711 return np.diag(F)

714def information_gain(x, y): 

715 """Calculate mutual information for each feature by formula 

716 I(X,Y) = H(Y) - H(Y|X). Bigger values mean more important features. This 

717 measure works best with discrete features due to being based on information 

718 theory. 

719 

720 Parameters 

721 ---------- 

722 x : array-like, shape (n_samples, n_features) 

723 The training input samples. 

724 y : array-like, shape (n_samples,) 

725 The target values. 

726 

727 Returns 

728 ------- 

729 array-like, shape (n_features,) : feature scores 

730 

731 See Also 

732 -------- 

733 

734 Examples 

735 -------- 

736 >>> from ITMO_FS.filters.univariate import information_gain 

737 >>> import numpy as np 

738 >>> from sklearn.preprocessing import KBinsDiscretizer 

739 >>> x = np.array([[1, 2, 3, 3, 1], [2, 2, 3, 3, 2], [1, 3, 3, 1, 3], 

740 ... [3, 1, 3, 1, 4], [4, 4, 3, 1, 5]]) 

741 >>> y = np.array([1, 2, 3, 4, 5]) 

742 >>> est = KBinsDiscretizer(n_bins=10, encode='ordinal') 

743 >>> x = est.fit_transform(x) 

744 >>> information_gain(x, y) 

745 array([1.33217904, 1.33217904, 0. , 0.67301167, 1.60943791]) 

746 """ 

747 entropy_x = entropy(y) 

748 cond_entropy = np.apply_along_axis(conditional_entropy, 0, x, y) 

749 return entropy_x - cond_entropy

752def anova(x, y): 

753 """Calculate anova measure for each feature. Bigger values mean more 

754 important features. 

755 

756 Parameters 

757 ---------- 

758 x : array-like, shape (n_samples, n_features) 

759 The training input samples. 

760 y : array-like, shape (n_samples,) 

761 The target values. 

762 

763 Returns 

764 ------- 

765 array-like, shape (n_features,) : feature scores 

766 

767 See Also 

768 -------- 

769 Lowry, Richard. "Concepts and Applications of Inferential Statistics". 

770 Chapter 14. http://vassarstats.net/textbook/ 

771 

772 Note: 

773 The Anova score is counted for checking hypothesis if variances of two 

774 samples are similar, this measure only returns you counted F-score. 

775 For understanding whether samples' variances are similar you should 

776 compare recieved result with value of F-distribution function, for 

777 example use: 

778 https://docs.scipy.org/doc/scipy/reference/generated/scipy.special.fdtrc.html#scipy.special.fdtrc 

779 

780 Examples 

781 -------- 

782 >>> from ITMO_FS.filters.univariate import anova 

783 >>> import numpy as np 

784 >>> x = np.array([[1, 2, 3, 3, 1], [2, 2, 3, 3, 2], [1, 3, 3, 1, 3], 

785 ... [3, 1, 3, 1, 4], [4, 4, 3, 1, 5]]) 

786 >>> y = np.array([1, 2, 1, 3, 3]) 

787 >>> anova(x, y) 

788 array([12.6 , 0.04, nan, 1.4 , 3. ]) 

789 """ 

790 split_by_class = [x[y == k] for k in np.unique(y)] 

791 num_classes = len(np.unique(y)) 

792 num_samples = x.shape[0] 

793 num_samples_by_class = [s.shape[0] for s in split_by_class] 

794 sq_sum_all = sum((s ** 2).sum(axis=0) for s in split_by_class) 

795 sum_group = [np.asarray(s.sum(axis=0)) for s in split_by_class] 

796 sq_sum_combined = sum(sum_group) ** 2 

797 sum_sq_group = [np.asarray((s ** 2).sum(axis=0)) for s in split_by_class] 

798 sq_sum_group = [s ** 2 for s in sum_group] 

799 sq_sum_total = sq_sum_all - sq_sum_combined / float(num_samples) 

800 sq_sum_within = sum( 

801 [sum_sq_group[i] - sq_sum_group[i] / num_samples_by_class[i] for i in 

802 range(num_classes)]) 

803 sq_sum_between = sq_sum_total - sq_sum_within 

804 deg_free_between = num_classes - 1 

805 deg_free_within = num_samples - num_classes 

806 ms_between = sq_sum_between / float(deg_free_between) 

807 ms_within = sq_sum_within / float(deg_free_within) 

808 f = ms_between / ms_within 

809 return np.array(f)

812def modified_t_score(x, y): 

813 """Calculate the Modified T-score for each feature. Bigger values mean 

814 more important features. 

815 

816 Parameters 

817 ---------- 

818 x : array-like, shape (n_samples, n_features) 

819 The input samples. 

820 y : array-like, shape (n_samples,) 

821 The classes for the samples. There can be only 2 classes. 

822 

823 Returns 

824 ------- 

825 array-like, shape (n_features,) : feature scores 

826 

827 See Also 

828 -------- 

829 For more details see paper <https://dergipark.org.tr/en/download/article-file/261247>. 

830 

831 Examples 

832 -------- 

833 >>> from ITMO_FS.filters.univariate import modified_t_score 

834 >>> import numpy as np 

835 >>> x = np.array([[3, 3, 3, 2, 2], [3, 3, 1, 2, 3], [1, 3, 5, 1, 1], 

836 ... [3, 1, 4, 3, 1], [3, 1, 2, 3, 1]]) 

837 >>> y = np.array([1, 1, 2, 1, 2]) 

838 >>> modified_t_score(x, y) 

839 array([1.68968099, 0.12148022, 0.39653932, 0.17682997, 2.04387142]) 

840 """ 

841 classes = np.unique(y) 

842 

843 size_class0 = y[y == classes[0]].size 

844 size_class1 = y[y == classes[1]].size 

845 

846 mean_class0 = np.mean(x[y == classes[0]], axis=0) 

847 mean_class0 = np.nan_to_num(mean_class0) 

848 mean_class1 = np.mean(x[y == classes[1]], axis=0) 

849 mean_class1 = np.nan_to_num(mean_class1) 

850 

851 std_class0 = np.std(x[y == classes[0]], axis=0) 

852 std_class0 = np.nan_to_num(std_class0) 

853 std_class1 = np.std(x[y == classes[1]], axis=0) 

854 std_class1 = np.nan_to_num(std_class1) 

855 

856 corr_with_y = np.apply_along_axis( 

857 lambda feature: abs(np.corrcoef(feature, y)[0][1]), 0, x) 

858 corr_with_y = np.nan_to_num(corr_with_y) 

859 

860 corr_with_others = abs(np.corrcoef(x, rowvar=False)) 

861 corr_with_others = np.nan_to_num(corr_with_others) 

862 

863 mean_of_corr_with_others = ( 

864 corr_with_others.sum(axis=1) 

865 - corr_with_others.diagonal()) / (len(corr_with_others) - 1) 

866 

867 t_score_numerator = abs(mean_class0 - mean_class1) 

868 t_score_denominator = np.sqrt( 

869 (size_class0 * np.square(std_class0) + size_class1 * np.square( 

870 std_class1)) / (size_class0 + size_class1)) 

871 modificator = corr_with_y / mean_of_corr_with_others 

872 

873 modified_t_score = t_score_numerator / t_score_denominator * modificator 

874 modified_t_score = np.nan_to_num(modified_t_score) 

875 

876 return modified_t_score

9def conditional_entropy(x_j, y): 

10 """Calculate the conditional entropy (H(Y|X)) between two arrays. 

11 

12 Parameters 

13 ---------- 

14 x_j : array-like, shape (n,) 

15 The first array. 

16 y : array-like, shape (n,) 

17 The second array. 

18 

19 Returns 

20 ------- 

21 float : H(Y|X) value 

22 

23 Examples 

24 -------- 

25 >>> from ITMO_FS.utils.information_theory import conditional_entropy 

26 >>> conditional_entropy([1,2,1,3,4], [1,2,3,4,5]) 

27 0.2772588722239781 

28 >>> conditional_entropy([1], [2]) 

29 0.0 

30 >>> conditional_entropy([1,2,1,3,2,4], [3,3,3,3,3,3]) 

31 0.0 

32 >>> conditional_entropy([1,2,3,1,3,2,3,4,1], [1,2,1,3,1,4,4,1,5]) 

33 0.7324081924454064 

34 """ 

35 buf = [[e[1] for e in g] for _, g in 

36 groupby(sorted(zip(x_j, y)), itemgetter(0))] 

37 return fsum(entropy(group) * len(group) for group in buf) / len(x_j)

10class IWSSr_SFLA(BaseWrapper): 

11 """IWSSr-SFLA (Incremental Wrapper Subset Selection with replacement and 

12 Shuffled Frog Leaping Algorithm). 

13 

14 Parameters 

15 ---------- 

16 estimator : object 

17 A supervised learning estimator that should have a fit(X, y) method and 

18 a predict(X) method. 

19 measure_iwssr : string or callable 

20 A standard estimator metric (e.g. 'f1' or 'roc_auc') or a callable with 

21 signature measure(estimator, X, y) which should return only a single 

22 value that would be used in the IWSSr algorithm. 

23 measure_frogs : string or callable 

24 A standard estimator metric (e.g. 'f1' or 'roc_auc') or a callable with 

25 signature measure(estimator, X, y) which should return only a single 

26 value that would be used to measure the frog's fitness value. 

27 cv : int 

28 Number of folds in cross-validation. 

29 relief_iterations : int 

30 Amount of iterations to do in the Relief algorithm. 

31 sfla_m : int 

32 Amount of memplexes in the SFLA algorithm. 

33 sfla_n : int 

34 Amount of frogs in each memplex in the SFLA algorithm. 

35 sfla_q : int 

36 Amount of frogs in each submemplex in the SFLA algorithm. Should be 

37 lower than sfla_n. 

38 s : int 

39 Maximum amount of features that can change during one leap in the SFLA 

40 algorithm. 

41 iterations : int 

42 Total iteration number in the SFLA algorithm. 

43 iterations_leaps : int 

44 Total amount of leaps that each memplex would try to do during each 

45 iteration. 

46 seed : int 

47 Python random seed. 

48 

49 See Also 

50 -------- 

51 For more details see `this paper <https://www.nature.com/articles/s41598-019-54987-1/>`_. 

52 

53 Examples 

54 -------- 

55 >>> from ITMO_FS.hybrid import IWSSr_SFLA 

56 >>> from sklearn.datasets import make_classification 

57 >>> from sklearn.preprocessing import KBinsDiscretizer 

58 >>> from sklearn.linear_model import LogisticRegression 

59 >>> dataset = make_classification(n_samples=100, n_features=20, 

60 ... n_informative=5, n_redundant=0, shuffle=False, random_state=42) 

61 >>> x, y = np.array(dataset[0]), np.array(dataset[1]) 

62 >>> x = KBinsDiscretizer(n_bins=10, encode='ordinal', 

63 ... strategy='uniform').fit_transform(x) 

64 >>> algo = IWSSr_SFLA(LogisticRegression()).fit(x, y) 

65 >>> algo.selected_features_ 

66 array([ 1, 3, 4, 10, 13, 15, 17], dtype=int64) 

67 """

85 def __generate_frog(self, weights): 

86 """Generate a frog from the given feature weights. 

87 

88 Parameters 

89 ---------- 

90 weights : array-like, shape (n_features,) 

91 Probabilities for the features. 

92 

93 Returns 

94 ------- 

95 array-like, shape (n_features,) : generated frog in the form of a 

96 boolean array 

97 """ 

98 f_num = self._rng.integers(self.n_features_) + 1 

99 features = self._rng.choice( 

100 self.n_features_, size=f_num, replace=False, p=weights) 

101 frog = np.full(self.n_features_, False) 

102 frog[features] = True 

103 return frog

119 def __iwssr(self, X, y, frog): 

120 """Perform the IWSSr algorithm for given data. 

121 

122 Parameters 

123 ---------- 

124 X : array-like, shape (n_samples, n_features) 

125 The training input samples. 

126 y : array-like, shape (n_samples,) 

127 The target values. 

128 frog : array-like, shape (n_features,) 

129 A frog (boolean array) of selected features. 

130 

131 Returns 

132 ------- 

133 array-like, shape (n_selected_features,) : selected features 

134 """ 

135 def __check_subset(subset): 

136 nonlocal best_measure 

137 nonlocal best_iteration_set 

138 

139 measure = cross_val_score( 

140 self._estimator, X[:, subset], y, cv=self.cv, 

141 scoring=self.measure_iwssr).mean() 

142 if measure > best_measure: 

143 best_measure = measure 

144 best_iteration_set = subset 

145 

146 frog_features = np.flatnonzero(frog) 

147 getLogger(__name__).info("Performing IWSSr for frog %s", frog_features) 

148 

149 weights = su_measure(X, y) 

150 getLogger(__name__).info("Feature weights: %s", weights) 

151 order = np.argsort(weights)[::-1] 

152 selected_features = np.array([order[0]]) 

153 best_measure = cross_val_score( 

154 self._estimator, X[:, selected_features], y, cv=self.cv, 

155 scoring=self.measure_iwssr).mean() 

156 best_iteration_set = selected_features 

157 getLogger(__name__).info( 

158 "Optimal feature set: %s, score: %d", 

159 best_iteration_set, best_measure) 

160 

161 for feature in order[1:]: 

162 getLogger(__name__).info( 

163 "Trying to add feature %d into the set", feature) 

164 if selected_features.shape[0] != 1: 

165 for i in range(selected_features.shape[0]): 

166 __check_subset( 

167 np.append( 

168 np.delete(selected_features, i), 

169 feature)) 

170 __check_subset(np.append(selected_features, feature)) 

171 getLogger(__name__).info( 

172 "Optimal feature set: %s, score: %d", 

173 best_iteration_set, best_measure) 

174 selected_features = best_iteration_set 

175 

176 selected_features = frog_features[selected_features] 

177 getLogger(__name__).info("New selected features: %s", selected_features) 

178 

179 new_frog = np.full(self.n_features_, False) 

180 new_frog[selected_features] = True 

181 return new_frog

183 def __create_submemplex(self, fitness_values): 

184 """Create a submemplex from the memplex of frogs. 

185 

186 Parameters 

187 ---------- 

188 fitness_values : array-like, shape (sfla_n,) 

189 Fitness values for all frogs in the memplex. 

190 

191 Returns 

192 ------- 

193 array-like, shape (sfla_q,) : indices of frogs in the submemplex 

194 """ 

195 order = np.argsort(fitness_values)[::-1] 

196 weights = np.zeros(self.sfla_n) 

197 weights[order] = np.arange(self.sfla_n) 

198 weights = (2 * (self.sfla_n - weights) 

199 / (self.sfla_n * (self.sfla_n + 1))) 

200 return self._rng.choice( 

201 self.sfla_n, size=self.sfla_q, replace=False, p=weights)

203 def __leap(self, frogs, fitness_values, memplex, submemplex, X, y, 

204 relief_weights): 

205 """Create a new worst frog in the submemplex by trying to make it 

206 leap towards better frogs. 

207 

208 Parameters 

209 ---------- 

210 frogs : array-like, shape (sfla_n * sfla_m, n_features) 

211 The population of frogs. 

212 fitness_values : array-like, shape (sfla_n * sfla_m) 

213 Fitness value for each frog. 

214 memplex : array-like, shape (sfla_n,) 

215 Indices of frogs in the memplex. 

216 submemplex : array-like, shape (sfla_q,) 

217 Indices of frogs in the submemplex. 

218 X : array-like, shape (n_samples, n_features) 

219 The training input samples. 

220 y : array-like, shape (n_samples,) 

221 The target values. 

222 relief_weights : array-like, shape (n_features,) 

223 Weights for the features. 

224 

225 Returns 

226 ------- 

227 tuple (int, array-like, float) : index of the worst frog in the 

228 submemplex, the new worst frog and its fitness value 

229 """ 

230 memplex_best_frog = memplex[np.argsort(fitness_values[memplex])[-1]] 

231 submemplex_worst_frog = submemplex[np.argsort( 

232 fitness_values[submemplex])[0]] 

233 getLogger(__name__).info( 

234 "Trying to evolve frog %s", submemplex_worst_frog) 

235 worst_frog_fitness = fitness_values[submemplex_worst_frog] 

236 

237 new_worst_frog = self.__leap_towards( 

238 frogs[memplex_best_frog], frogs[submemplex_worst_frog], X, y) 

239 new_worst_frog_fitness = cross_val_score( 

240 self._estimator, X[:, new_worst_frog], y, cv=self.cv, 

241 scoring=self.measure_frogs).mean() 

242 getLogger(__name__).info( 

243 "After leaping to submemplex best frog: frog = %s, fitness = %d", 

244 new_worst_frog, new_worst_frog_fitness) 

245 if new_worst_frog_fitness > worst_frog_fitness: 

246 return submemplex_worst_frog, new_worst_frog, new_worst_frog_fitness 

247 

248 population_best_frog = np.argsort(fitness_values)[-1] 

249 new_worst_frog = self.__leap_towards( 

250 frogs[population_best_frog], frogs[submemplex_worst_frog], X, y) 

251 new_worst_frog_fitness = cross_val_score( 

252 self._estimator, X[:, new_worst_frog], y, cv=self.cv, 

253 scoring=self.measure_frogs).mean() 

254 getLogger(__name__).info( 

255 "After leaping to memplex best frog: frog = %s, fitness = %d", 

256 new_worst_frog, new_worst_frog_fitness) 

257 if new_worst_frog_fitness > worst_frog_fitness: 

258 return submemplex_worst_frog, new_worst_frog, new_worst_frog_fitness 

259 

260 new_worst_frog = self.__generate_frog(relief_weights) 

261 new_worst_frog_fitness = cross_val_score( 

262 self._estimator, X[:, new_worst_frog], y, cv=self.cv, 

263 scoring=self.measure_frogs).mean() 

264 getLogger(__name__).info( 

265 "Replacing with random frog: frog = %s, fitness = %d", 

266 new_worst_frog, new_worst_frog_fitness) 

267 return submemplex_worst_frog, new_worst_frog, new_worst_frog_fitness

269 def __leap_towards(self, frog_to, frog_from, X, y): 

270 """Perform a leap from the worse frog to the better frog. 

271 

272 Parameters 

273 ---------- 

274 frog_to : array-like, shape (n_features,) 

275 The better frog (one to leap to). 

276 frog_from : array-like, shape (n_features,) 

277 The worse frog (one that leaps). 

278 X : array-like, shape (n_samples, n_features) 

279 The training input samples. 

280 y : array-like, shape (n_samples,) 

281 The target values. 

282 

283 Returns 

284 ------- 

285 array-like, shape (n_features,) : the new worst frog 

286 """ 

287 len_to = np.count_nonzero(frog_to) 

288 len_from = np.count_nonzero(frog_from) 

289 if len_to > len_from: 

290 to_add = frog_to & np.logical_not(frog_from) 

291 weights = su_measure(X[:, to_add], y) 

292 weights /= np.sum(weights) 

293 features_num = min( 

294 min( 

295 int(self._rng.random() * (len_to - len_from)), 

296 self.s), 

297 np.count_nonzero(to_add)) 

298 added_features = self._rng.choice( 

299 np.flatnonzero(to_add), size=features_num, replace=False, 

300 p=weights) 

301 new_worst_frog = np.copy(frog_from) 

302 new_worst_frog[added_features] = True 

303 else: 

304 weights = su_measure(X[:, frog_from], y) 

305 weights /= np.sum(weights) 

306 features_num = min( 

307 min( 

308 int(self._rng.random() * (len_from - len_to)), 

309 self.s), 

310 len_from - 1) 

311 left_features = self._rng.choice( 

312 np.flatnonzero(frog_from), size=len_from - features_num, 

313 replace=False, p=weights) 

314 new_worst_frog = np.full(self.n_features_, False) 

315 new_worst_frog[left_features] = True 

316 return new_worst_frog

318 def __apply_sfla(self, frogs, X, y, relief_weights): 

319 """Apply the SFLA algorithm to a population of frogs. 

320 

321 Parameters 

322 ---------- 

323 frogs : array-like, shape (sfla_n * sfla_m, n_features) 

324 The population of frogs. 

325 X : array-like, shape (n_samples, n_features) 

326 The training input samples. 

327 y : array-like, shape (n_samples,) 

328 The target values. 

329 relief_weights : array-like, shape (n_features,) 

330 Weights for the features. 

331 

332 Returns 

333 ------- 

334 array-like, shape (sfla_n * sfla_m, n_features) : the evolved population 

335 of frogs 

336 """ 

337 fitness_values = np.vectorize( 

338 lambda frog: cross_val_score( 

339 self._estimator, X[:, frog], y, cv=self.cv, 

340 scoring=self.measure_frogs).mean(), 

341 signature='(1)->()')(frogs) 

342 getLogger(__name__).info("Fitness values for frogs: %s", fitness_values) 

343 for _ in range(self.iterations): 

344 memplexes = self.__create_memplexes() 

345 for memplex in memplexes: 

346 for _ in range(self.iterations_leaps): 

347 submemplex = memplex[self.__create_submemplex( 

348 fitness_values[memplex])] 

349 (worst_frog_index, new_worst_frog, 

350 new_worst_frog_fitness) = self.__leap( 

351 frogs, fitness_values, memplex, submemplex, X, y, 

352 relief_weights) 

353 frogs[worst_frog_index] = new_worst_frog 

354 fitness_values[worst_frog_index] = new_worst_frog_fitness 

355 return frogs

360

372 self._rng = np.random.default_rng(self.seed)

395 scoring=self.measure_frogs).mean(),
